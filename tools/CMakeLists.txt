# Project tools binaries dir
SET(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${PROJECT_SOURCE_DIR}/tools/bin)

INCLUDE_DIRECTORIES(SYSTEM ${VTK_INCLUDE_DIRS})                                 
SET_PROPERTY(DIRECTORY APPEND PROPERTY COMPILE_DEFINITIONS ${VTK_DEFINITIONS}) 

SET(LDIR_TARGET_NAME "light_directions")
ADD_EXECUTABLE(${LDIR_TARGET_NAME} 
    ${CMAKE_CURRENT_SOURCE_DIR}/light-directions/main.cpp)
SET_TARGET_PROPERTIES(${LDIR_TARGET_NAME} PROPERTIES LINKER_LANGUAGE CXX)
SET_TARGET_PROPERTIES(${LDIR_TARGET_NAME} PROPERTIES CXX_STANDARD 11)
SET_TARGET_PROPERTIES(${LDIR_TARGET_NAME} PROPERTIES CXX_STANDARD_REQUIRED ON)
TARGET_LINK_LIBRARIES(${LDIR_TARGET_NAME} 
    ${OpenCV_LIBS} 
    ${Boost_LIBRARIES}
    ${VTK_LIBRARIES}
    ${PROJECT_NAME}_common
    ${PROJECT_NAME}_io
    ${PROJECT_NAME}_filter
    ${PROJECT_NAME}_render)

SET(CAMS_TARGET_NAME "show_cameras")

    ${CMAKE_CURRENT_SOURCE_DIR}/show-cameras/main.cpp)

SET_TARGET_PROPERTIES(${CAMS_TARGET_NAME} PROPERTIES CXX_STANDARD 11)
SET_TARGET_PROPERTIES(${CAMS_TARGET_NAME} PROPERTIES CXX_STANDARD_REQUIRED ON)
TARGET_LINK_LIBRARIES(${CAMS_TARGET_NAME} 
    ${OpenCV_LIBS}

    ${VTK_LIBRARIES}
    ${PROJECT_NAME}_common
    ${PROJECT_NAME}_io
